Voici le principe de fonctionnement :

    Importer la clé publique du serveur (fichier .pem).

    Générer une clé ECC éphémère côté client.

    Faire un ECDH entre la clé éphémère et la clé publique du serveur → ça te donne une clé partagée.

    Deriver une clé AES à partir de cette clé partagée (via SHA256 par exemple).

    Chiffrer ton message avec AES-GCM ou AES-CTR.

    Envoyer au serveur : {clé publique éphémère, IV, ciphertext}.

🛠️ Exemple simplifié en C (OpenSSL)

#include <openssl/ec.h>
#include <openssl/evp.h>
#include <openssl/pem.h>
#include <openssl/sha.h>
#include <openssl/rand.h>

// 1. Charger la clé publique du serveur
EC_KEY *load_server_pubkey(const char *path) {
    FILE *fp = fopen(path, "r");
    if (!fp) return NULL;
    EC_KEY *key = PEM_read_EC_PUBKEY(fp, NULL, NULL, NULL);
    fclose(fp);
    return key;
}

// 2. Générer une clé éphémère
EC_KEY *generate_ephemeral_key() {
    EC_KEY *key = EC_KEY_new_by_curve_name(NID_X9_62_prime256v1);
    EC_KEY_generate_key(key);
    return key;
}

// 3. Calculer la clé partagée
int derive_shared_secret(EC_KEY *client_key, EC_KEY *server_pub, unsigned char *secret) {
    const EC_POINT *server_point = EC_KEY_get0_public_key(server_pub);
    const EC_GROUP *group = EC_KEY_get0_group(client_key);
    return ECDH_compute_key(secret, 32, server_point, client_key, NULL);
}

// 4. Deriver une clé AES
void sha256(const unsigned char *input, size_t len, unsigned char *output) {
    SHA256(input, len, output);
}


Tu peux ensuite utiliser EVP_EncryptInit_ex() pour chiffrer ton message avec AES-256-GCM ou AES-CTR, en utilisant la clé dérivée.
📦 Format du message à envoyer

{
  ephemeral_pub_key (PEM ou DER),
  iv,
  ciphertext,
  tag (si AES-GCM)
}

Le serveur peut alors :

    Extraire la clé éphémère.

    Faire le même ECDH avec sa clé privée.

    Deriver la même clé AES.

    Déchiffrer le message.



func handleECDHHandshake(conn net.Conn) {
    defer conn.Close()

    // 1. Lire la clé publique du client
    clientPubBytes := make([]byte, 91) // taille typique d'une clé EC P256 en PEM
    _, err := conn.Read(clientPubBytes)
    if err != nil {
        log.Println("Erreur lecture clé client:", err)
        return
    }

    block, _ := pem.Decode(clientPubBytes)
    pubKeyInterface, _ := x509.ParsePKIXPublicKey(block.Bytes)
    clientPub := pubKeyInterface.(*ecdsa.PublicKey)

    // 2. Générer clé serveur
    serverPriv, _ := ecdsa.GenerateKey(elliptic.P256(), rand.Reader)
    serverPubBytes, _ := x509.MarshalPKIXPublicKey(&serverPriv.PublicKey)
    serverPem := pem.EncodeToMemory(&pem.Block{Type: "PUBLIC KEY", Bytes: serverPubBytes})

    // 3. Envoyer la clé publique serveur
    conn.Write(serverPem)

    // 4. Calculer secret partagé
    x, _ := serverPriv.PublicKey.Curve.ScalarMult(clientPub.X, clientPub.Y, serverPriv.D.Bytes())
    sharedSecret := sha256.Sum256(x.Bytes())

    // Tu peux maintenant utiliser sharedSecret[:] comme clé AES
}




#include <openssl/ec.h>
#include <openssl/pem.h>
#include <openssl/sha.h>
#include <openssl/aes.h>
#include <stdio.h>
#include <string.h>
#include <arpa/inet.h>
#include <unistd.h>

int main() {
    // 1. Générer paire EC
    EC_KEY *client_key = EC_KEY_new_by_curve_name(NID_X9_62_prime256v1);
    EC_KEY_generate_key(client_key);

    // 2. Extraire clé publique
    BIO *bio = BIO_new(BIO_s_mem());
    PEM_write_bio_EC_PUBKEY(bio, client_key);
    char pubkey[512];
    int len = BIO_read(bio, pubkey, sizeof(pubkey));

    // 3. Envoyer au serveur
    int sock = socket(AF_INET, SOCK_STREAM, 0);
    struct sockaddr_in serv_addr = {.sin_family = AF_INET, .sin_port = htons(5437)};
    inet_pton(AF_INET, "127.0.0.1", &serv_addr.sin_addr);
    connect(sock, (struct sockaddr*)&serv_addr, sizeof(serv_addr));
    send(sock, pubkey, len, 0);

    // 4. Recevoir clé publique serveur
    char server_pub[512];
    int recv_len = recv(sock, server_pub, sizeof(server_pub), 0);
    BIO *bio_srv = BIO_new_mem_buf(server_pub, recv_len);
    EC_KEY *server_key = PEM_read_bio_EC_PUBKEY(bio_srv, NULL, NULL, NULL);

    // 5. Calculer secret partagé
    const EC_POINT *srv_point = EC_KEY_get0_public_key(server_key);
    const EC_GROUP *group = EC_KEY_get0_group(client_key);
    unsigned char secret[32];
    ECDH_compute_key(secret, sizeof(secret), srv_point, client_key, NULL);

    // 6. Deriver clé AES
    unsigned char aes_key[32];
    SHA256(secret, sizeof(secret), aes_key);

    printf("Clé AES partagée : ");
    for (int i = 0; i < 32; i++) printf("%02x", aes_key[i]);
    printf("\n");

    // Tu peux maintenant chiffrer avec AES
}
